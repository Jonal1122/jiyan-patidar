
CREATE TABLE influencers (
    influencer_id INT PRIMARY KEY,
    name VARCHAR(100)
);
CREATE TABLE posts (
    post_id INT PRIMARY KEY,
    influencer_id INT,
    caption VARCHAR(255),
    FOREIGN KEY (influencer_id) REFERENCES influencers(influencer_id)
);
